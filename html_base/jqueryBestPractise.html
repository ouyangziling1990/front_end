1. jquery 少改动dom结构

	-- normal
	var ele = $("#elem")
	ele.data(key, value)

	--- better

	var elem = $("#elem")
	$.data(elem[0], key, value)

2. 使用Pub/Sub模式管理事件

	--- normal
	function doSomthing(){
		doSomethingElse()
		doOneMoreThing()
	
	}
	--- better
	function doSomething(){
		$.trigger("DO_SOMETHING_DONE")
	}
	$(document).on("DO_SOMETHING_DONE", function(){
		doSomethingElse()
	})
